	package main

import "fmt"

func Somaate(n int) int {
	var soma int
	for i := 1; i <= n; i++ {
		soma += i
	}
	return soma
}

func main() {
	var n int
	fmt.Println("Digite um numero: ")
	fmt.Scan(&n)
	fmt.Println(Somaate(n))
}
